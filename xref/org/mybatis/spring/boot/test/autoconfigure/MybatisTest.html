39  /**
40   * Annotation that can be used in combination with {@code @RunWith(SpringRunner.class)}
41   * for a typical mybatis test. Can be used when a test focuses <strong>only</strong> on
42   * mybatis-based components.
43   * <p>
44   * Using this annotation will disable full auto-configuration and instead apply only
45   * configuration relevant to mybatis tests.
46   * <p>
47   * By default, tests annotated with {@code @JdbcTest} will use an embedded in-memory
48   * database (replacing any explicit or usually auto-configured DataSource). The
49   * {@link AutoConfigureTestDatabase @AutoConfigureTestDatabase} annotation can be used to
50   * override these settings.
51   * <p>
52   * If you are looking to load your full application configuration, but use an embedded
53   * database, you should consider {@link SpringBootTest @SpringBootTest} combined with
54   * {@link AutoConfigureTestDatabase @AutoConfigureTestDatabase} rather than this
55   * annotation.
56   *
57   * @author wonwoo
58   * @see AutoConfigureMybatis
59   * @since 1.2.1
60   */
61  @Target(ElementType.TYPE)
62  @Retention(RetentionPolicy.RUNTIME)
63  @Documented
64  @Inherited
65  @BootstrapWith(SpringBootTestContextBootstrapper.class)
66  @OverrideAutoConfiguration(enabled = false)
67  @TypeExcludeFilters(MybatisTypeExcludeFilter.class)
68  @Transactional
69  @AutoConfigureCache
70  @AutoConfigureMybatis
71  @AutoConfigureTestDatabase
72  @ImportAutoConfiguration
73  public @interface MybatisTest {
74  
75    /**
76     * Determines if default filtering should be used with
77     * {@link SpringBootApplication @SpringBootApplication}. By default no beans are
78     * included.
79     *
80     * @return if default filters should be used
81     * @see #includeFilters()
82     * @see #excludeFilters()
83     */
84    boolean useDefaultFilters() default true;
85  
86    /**
87     * A set of include filters which can be used to add otherwise filtered beans to the
88     * application context.
89     *
90     * @return include filters to apply
91     */
92    Filter[] includeFilters() default {};
93  
94    /**
95     * A set of exclude filters which can be used to filter beans that would otherwise be
96     * added to the application context.
97     *
98     * @return exclude filters to apply
99     */
100   Filter[] excludeFilters() default {};
101 
102   /**
103    * Auto-configuration exclusions that should be applied for this test.
104    * @return auto-configuration exclusions to apply
105    */
106   @AliasFor(annotation = ImportAutoConfiguration.class, attribute = "exclude")
107   Class<?>[] excludeAutoConfiguration() default {};
108 
109 }
