210   public Resource[] resolveMapperLocations() {
211     return Stream.of(Optional.ofNullable(this.mapperLocations).orElse(new String[0]))
212         .flatMap(location -> Stream.of(getResources(location))).toArray(Resource[]::new);
213   }
214 
215   private Resource[] getResources(String location) {
216     try {
217       return resourceResolver.getResources(location);
218     } catch (IOException e) {
219       return new Resource[0];
220     }
221   }

679       mapper.from(getMultipleResultSetsEnabled()).to(target::setMultipleResultSetsEnabled);
680       mapper.from(getUseGeneratedKeys()).to(target::setUseGeneratedKeys);
681       mapper.from(getUseColumnLabel()).to(target::setUseColumnLabel);
682       mapper.from(getCacheEnabled()).to(target::setCacheEnabled);
683       mapper.from(getCallSettersOnNulls()).to(target::setCallSettersOnNulls);
684       mapper.from(getUseActualParamName()).to(target::setUseActualParamName);
685       mapper.from(getReturnInstanceForEmptyRow()).to(target::setReturnInstanceForEmptyRow);
686       mapper.from(getShrinkWhitespacesInSql()).to(target::setShrinkWhitespacesInSql);
687       mapper.from(getNullableOnForEach()).to(target::setNullableOnForEach);
688       mapper.from(getArgNameBasedConstructorAutoMapping()).to(target::setArgNameBasedConstructorAutoMapping);
689       mapper.from(getLazyLoadingEnabled()).to(target::setLazyLoadingEnabled);
690       mapper.from(getLogPrefix()).to(target::setLogPrefix);
691       mapper.from(getLazyLoadTriggerMethods()).to(target::setLazyLoadTriggerMethods);
692       mapper.from(getDefaultStatementTimeout()).to(target::setDefaultStatementTimeout);
693       mapper.from(getDefaultFetchSize()).to(target::setDefaultFetchSize);
694       mapper.from(getLocalCacheScope()).to(target::setLocalCacheScope);
695       mapper.from(getJdbcTypeForNull()).to(target::setJdbcTypeForNull);
696       mapper.from(getDefaultResultSetType()).to(target::setDefaultResultSetType);
697       mapper.from(getDefaultExecutorType()).to(target::setDefaultExecutorType);
698       mapper.from(getAutoMappingBehavior()).to(target::setAutoMappingBehavior);
699       mapper.from(getAutoMappingUnknownColumnBehavior()).to(target::setAutoMappingUnknownColumnBehavior);
700       mapper.from(getVariables()).to(target::setVariables);
701       mapper.from(getLogImpl()).to(target::setLogImpl);
702       mapper.from(getVfsImpl()).to(target::setVfsImpl);
