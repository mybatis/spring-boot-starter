172   public Resource[] resolveMapperLocations() {
173     ResourcePatternResolver resourceResolver = new PathMatchingResourcePatternResolver();
174     List<Resource> resources = new ArrayList<Resource>();
175     if (this.mapperLocations != null) {
176       for (String mapperLocation : this.mapperLocations) {
177         try {
178           Resource[] mappers = resourceResolver.getResources(mapperLocation);
179           resources.addAll(Arrays.asList(mappers));
180         } catch (IOException e) {
181           // ignore
182         }
183       }
184     }
185     return resources.toArray(new Resource[resources.size()]);
186   }
