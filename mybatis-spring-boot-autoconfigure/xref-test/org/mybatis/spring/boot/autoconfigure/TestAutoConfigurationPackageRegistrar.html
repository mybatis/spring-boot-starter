16  package org.mybatis.spring.boot.autoconfigure;
17  
18  import org.springframework.beans.factory.support.BeanDefinitionRegistry;
19  import org.springframework.boot.autoconfigure.AutoConfigurationPackages;
20  import org.springframework.context.annotation.ImportBeanDefinitionRegistrar;
21  import org.springframework.core.annotation.AnnotationAttributes;
22  import org.springframework.core.type.AnnotationMetadata;
23  import org.springframework.util.ClassUtils;
24  
25  /**
26   * {@link ImportBeanDefinitionRegistrar} to store the base package for tests.
27   *
28   * @author Eddú Meléndez
29   */
30  public class TestAutoConfigurationPackageRegistrar implements ImportBeanDefinitionRegistrar {
31  
32    @Override
33    public void registerBeanDefinitions(AnnotationMetadata metadata, BeanDefinitionRegistry registry) {
34      AnnotationAttributes attributes = AnnotationAttributes
35          .fromMap(metadata.getAnnotationAttributes(TestAutoConfigurationPackage.class.getName(), true));
36      AutoConfigurationPackages.register(registry, ClassUtils.getPackageName(attributes.getString("value")));
37    }
38  
39  }
