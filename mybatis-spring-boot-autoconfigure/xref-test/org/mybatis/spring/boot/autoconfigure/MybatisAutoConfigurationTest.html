73  	@Test
74  	public void testNoDataSource() throws Exception {
75  		this.context.register(MybatisAutoConfiguration.class,
76  				PropertyPlaceholderAutoConfiguration.class);
77  		this.context.refresh();
78  		assertEquals(0, this.context.getBeanNamesForType(SqlSessionFactory.class).length);
79  	}

81  	@Test
82  	public void testDefaultConfiguration() {
83  		this.context.register(EmbeddedDataSourceConfiguration.class,
84  				MybatisScanMapperConfiguration.class, MybatisAutoConfiguration.class,
85  				PropertyPlaceholderAutoConfiguration.class);
86  		this.context.refresh();
87  		assertEquals(1, this.context.getBeanNamesForType(SqlSessionFactory.class).length);
88  		assertEquals(1,
89  				this.context.getBeanNamesForType(SqlSessionTemplate.class).length);
90  		assertEquals(1, this.context.getBeanNamesForType(CityMapper.class).length);
91  	}

105 	@Test
106 	public void testWithTypeHandlersPackage() {
107 		EnvironmentTestUtils.addEnvironment(this.context,
108 				"mybatis.typeHandlersPackage:org.mybatis.spring.boot.autoconfigure.handler");
109 		this.context.register(EmbeddedDataSourceConfiguration.class,
110 				MybatisAutoConfiguration.class,
111 				PropertyPlaceholderAutoConfiguration.class);
112 		this.context.refresh();
113 
114 		TypeHandlerRegistry typeHandlerRegistry = this.context.getBean(SqlSessionFactory.class).getConfiguration().getTypeHandlerRegistry();
115 		assertTrue(typeHandlerRegistry.hasTypeHandler(BigInteger.class));
116 	}

121 				"mybatis.typeAliasesPackage:org.mybatis.spring.boot.autoconfigure.domain",
122 				"mybatis.mapperLocations:classpath:org/mybatis/spring/boot/autoconfigure/repository/CityMapper.xml");

124 				MybatisAutoConfiguration.class,

190 	@EnableAutoConfiguration

192 
193 		@Bean

195 			return new MyInterceptor();
196 		}
197 
198 	}

200 	@Intercepts(
201 			@Signature(type = Map.class, method = "get", args = { Object.class })
202 	)
203 	static class MyInterceptor implements Interceptor {
204 
205 		@Override
206 		public Object intercept(Invocation invocation) throws Throwable {
207 			return "Test";
208 		}
209 
210 		@Override
211 		public Object plugin(Object target) {
212 			return Plugin.wrap(target, this);
213 		}
214 
215 		@Override
216 		public void setProperties(Properties properties) {
217 
218 		}
219 	}
220 
221 	@Configuration
222 	static class DatabaseProvidersConfiguration {
223 
224 		@Bean
225 		public PropertiesFactoryBean vendorProperties() {
226 			Properties properties = new Properties();
227 			properties.put("SQL Server", "sqlserver");
228 			properties.put("DB2", "db2");
229 			properties.put("H2", "h2");
230 
231 			PropertiesFactoryBean propertiesFactoryBean = new PropertiesFactoryBean();
232 			propertiesFactoryBean.setProperties(properties);
233 			return propertiesFactoryBean;
234 		}
235 
236 		@Bean
237 		public VendorDatabaseIdProvider vendorDatabaseIdProvider(Properties vendorProperties) {
238 			VendorDatabaseIdProvider databaseIdProvider = new VendorDatabaseIdProvider();
239 			databaseIdProvider.setProperties(vendorProperties);
240 			return databaseIdProvider;
241 		}
