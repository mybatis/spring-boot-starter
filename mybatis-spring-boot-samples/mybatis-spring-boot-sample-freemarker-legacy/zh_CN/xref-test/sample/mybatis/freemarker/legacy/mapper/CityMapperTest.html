16  package sample.mybatis.freemarker.legacy.mapper;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  
20  import org.junit.jupiter.api.Test;
21  import org.mybatis.spring.boot.test.autoconfigure.MybatisTest;
22  import org.springframework.beans.factory.annotation.Autowired;
23  
24  import sample.mybatis.freemarker.legacy.domain.City;
25  
26  /**
27   * Tests for {@link CityMapper}.
28   *
29   * @author Kazuki Shimizu
30   */
31  @MybatisTest
32  class CityMapperTest {
33  
34    @Autowired
35    private CityMapper cityMapper;
36  
37    @Test
38    void findByState() {
39      City city = cityMapper.findByState("CA");
40      assertThat(city.getId()).isEqualTo(1);
41      assertThat(city.getName()).isEqualTo("San Francisco");
42      assertThat(city.getState()).isEqualTo("CA");
43      assertThat(city.getCountry()).isEqualTo("US");
44    }
45  
46    @Test
47    void findById() {
48      City city = cityMapper.findById(1L);
49      assertThat(city.getId()).isEqualTo(1);
50      assertThat(city.getName()).isEqualTo("San Francisco");
51      assertThat(city.getState()).isEqualTo("CA");
52      assertThat(city.getCountry()).isEqualTo("US");
53    }
54  
55    @Test
56    void findByName() {
57      City city = cityMapper.findByName("San Francisco");
58      assertThat(city.getId()).isEqualTo(1);
59      assertThat(city.getName()).isEqualTo("San Francisco");
60      assertThat(city.getState()).isEqualTo("CA");
61      assertThat(city.getCountry()).isEqualTo("US");
62    }
63  
64  }
