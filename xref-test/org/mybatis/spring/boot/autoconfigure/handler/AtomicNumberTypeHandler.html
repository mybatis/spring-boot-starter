43    }
44  
45    public Number getResult(CallableStatement cs, int columnIndex) throws SQLException {
46      return null;
47    }
48  
49    public Number getResult(ResultSet rs, int columnIndex) throws SQLException {
50      return null;
51    }
52  
53    @Override

56    }
57  
58  }
