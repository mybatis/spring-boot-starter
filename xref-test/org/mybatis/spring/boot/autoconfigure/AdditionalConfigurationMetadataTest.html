16  package org.mybatis.spring.boot.autoconfigure;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  
20  import com.jayway.jsonpath.DocumentContext;
21  import com.jayway.jsonpath.JsonPath;
22  
23  import java.io.IOException;
24  import java.util.List;
25  import java.util.Map;
26  
27  import org.junit.jupiter.api.Test;
28  import org.springframework.core.io.FileSystemResource;
29  
30  /**
31   * Tests for definition of additional-spring-configuration-metadata.json.
32   *
33   * @author Kazuki Shimizu
34   * @since 1.3.1
35   */
36  class AdditionalConfigurationMetadataTest {
37  
38    @Test
39    void testProperties() throws IOException {
40  
41      DocumentContext documentContext = JsonPath
42          .parse(new FileSystemResource("src/main/resources/META-INF/additional-spring-configuration-metadata.json")
43              .getInputStream());
44  
45      List<Map<String, Object>> properties = documentContext.read("$.properties");
46  
47      assertThat(properties.size()).isEqualTo(5);
48  
49      // assert for mybatis.configuration.default-scripting-language
50      {
51        Map<String, Object> element = properties.get(0);
52        assertThat(element.get("sourceType")).isEqualTo("org.apache.ibatis.session.Configuration");
53        assertThat(element.get("defaultValue")).isEqualTo("org.apache.ibatis.scripting.xmltags.XMLLanguageDriver");
54        assertThat(element.get("name")).isEqualTo("mybatis.configuration.default-scripting-language");
55        assertThat(element.get("type"))
56            .isEqualTo("java.lang.Class<? extends org.apache.ibatis.scripting.LanguageDriver>");
57        @SuppressWarnings("unchecked")
58        Map<String, Object> deprecation = (Map<String, Object>) element.get("deprecation");
59        assertThat(deprecation.get("reason")).isEqualTo(
60            "Because when this configuration property is used, there is case that custom language driver cannot be registered correctly.");
61        assertThat(deprecation.get("replacement")).isEqualTo("mybatis.default-scripting-language-driver");
62      }
63  
64      // assert for mybatis.configuration.default-enum-type-handler
65      {
66        Map<String, Object> element = properties.get(1);
67        assertThat(element.get("sourceType")).isEqualTo("org.apache.ibatis.session.Configuration");
68        assertThat(element.get("defaultValue")).isEqualTo("org.apache.ibatis.type.EnumTypeHandler");
69        assertThat(element.get("name")).isEqualTo("mybatis.configuration.default-enum-type-handler");
70        assertThat(element.get("type")).isEqualTo("java.lang.Class<? extends org.apache.ibatis.type.TypeHandler>");
71      }
72  
73      // assert for mybatis.lazy-initialization
74      {
75        Map<String, Object> element = properties.get(2);
76        assertThat(element.get("defaultValue")).isEqualTo(false);
77        assertThat(element.get("name")).isEqualTo("mybatis.lazy-initialization");
78        assertThat(element.get("type")).isEqualTo("java.lang.Boolean");
79      }
80  
81      // assert for mybatis.mapper-default-scope
82      {
83        Map<String, Object> element = properties.get(3);
84        assertThat(element.get("defaultValue")).isEqualTo("");
85        assertThat(element.get("name")).isEqualTo("mybatis.mapper-default-scope");
86        assertThat(element.get("type")).isEqualTo("java.lang.String");
87      }
88  
89      // assert for mybatis.scripting-language-driver.velocity.userdirective
90      {
91        Map<String, Object> element = properties.get(4);
92        assertThat(element.get("name")).isEqualTo("mybatis.scripting-language-driver.velocity.userdirective");
93        @SuppressWarnings("unchecked")
94        Map<String, Object> deprecation = (Map<String, Object>) element.get("deprecation");
95        assertThat(deprecation.get("level")).isEqualTo("error");
96        assertThat(deprecation.get("reason")).isEqualTo(
97            "The 'userdirective' is deprecated since Velocity 2.x. This property defined for keeping backward compatibility with older velocity version.");
98        assertThat(deprecation.get("replacement"))
99            .isEqualTo("mybatis.scripting-language-driver.velocity.velocity-settings.runtime.custom_directives");
100     }
101 
102   }
