46  /**
47   * {@code @CaptureSystemOutput} is a JUnit JUpiter extension for capturing output to {@code System.out} and
48   * {@code System.err} with expectations supported via Hamcrest matchers.
49   * <h4>Example Usage</h4>
50   *
51   * <pre style="code">
52   * {@literal @}Test
53   * {@literal @}CaptureSystemOutput
54   * void systemOut(OutputCapture outputCapture) {
55   *     outputCapture.expect(containsString("System.out!"));
56   *
57   *     System.out.println("Printed to System.out!");
58   * }
59   *
60   * {@literal @}Test
61   * {@literal @}CaptureSystemOutput
62   * void systemErr(OutputCapture outputCapture) {
63   *     outputCapture.expect(containsString("System.err!"));
64   *
65   *     System.err.println("Printed to System.err!");
66   * }
67   * </pre>
68   * <p>
69   * Based on code from Spring Boot's <a href=
70   * "https://github.com/spring-projects/spring-boot/blob/d3c34ee3d1bfd3db4a98678c524e145ef9bca51c/spring-boot-project/spring-boot-tools/spring-boot-test-support/src/main/java/org/springframework/boot/testsupport/rule/OutputCapture.java">OutputCapture</a>
71   * rule for JUnit 4 by Phillip Webb and Andy Wilkinson.
72   * <p>
73   * Borrowing source from Sam Brannen as listed online at spring and stackoverflow from here <a href=
74   * "https://github.com/sbrannen/junit5-demo/blob/master/src/test/java/extensions/CaptureSystemOutput.java">CaptureSystemOutput</a>
75   * <p>
76   * Additional changes to Sam Brannen logic supplied by kazuki43zoo from here <a href=
77   * "https://github.com/kazuki43zoo/mybatis-spring-boot/commit/317c9809326baba1f6ee0a0f8c2c471cd75993b3">enhancement
78   * capture</a>
79   *
80   * @author Sam Brannen
81   * @author Phillip Webb
82   * @author Andy Wilkinson
83   */
84  @Target({ TYPE, METHOD })
85  @Retention(RUNTIME)
86  @ExtendWith(CaptureSystemOutput.Extension.class)
87  public @interface CaptureSystemOutput {
88  
89    class Extension implements BeforeAllCallback, AfterAllCallback, AfterEachCallback, ParameterResolver {
90  
91      @Override
92      public void beforeAll(ExtensionContext context) {
93        getOutputCapture(context).captureOutput();
94      }
95  
96      public void afterAll(ExtensionContext context) {
97        getOutputCapture(context).releaseOutput();
98      }
99  
100     @Override
101     public void afterEach(ExtensionContext context) {
102       OutputCapture outputCapture = getOutputCapture(context);
103       try {
104         if (!outputCapture.matchers.isEmpty()) {
105           String output = outputCapture.toString();
106           assertThat(output, allOf(outputCapture.matchers));
107         }
108       } finally {
109         outputCapture.reset();
110       }
111     }
112 
113     @Override
114     public boolean supportsParameter(ParameterContext parameterContext, ExtensionContext extensionContext) {
115       boolean isTestMethodLevel = extensionContext.getTestMethod().isPresent();
116       boolean isOutputCapture = parameterContext.getParameter().getType() == OutputCapture.class;
117       return isTestMethodLevel && isOutputCapture;
118     }
119 
120     @Override
121     public Object resolveParameter(ParameterContext parameterContext, ExtensionContext extensionContext) {
122       return getOutputCapture(extensionContext);
123     }
124 
125     private OutputCapture getOutputCapture(ExtensionContext context) {
126       return getOrComputeIfAbsent(getStore(context), OutputCapture.class);
127     }
128 
129     private <V> V getOrComputeIfAbsent(Store store, Class<V> type) {
130       return store.getOrComputeIfAbsent(type, ReflectionSupport::newInstance, type);
131     }
132 
133     private Store getStore(ExtensionContext context) {
134       return context.getStore(Namespace.create(getClass()));
135     }
136 
137   }

139   /**
140    * {@code OutputCapture} captures output to {@code System.out} and {@code System.err}.
141    * <p>
142    * To obtain an instance of {@code OutputCapture}, declare a parameter of type {@code OutputCapture} in a JUnit
143    * Jupiter {@code @Test}, {@code @BeforeEach}, or {@code @AfterEach} method.
144    * <p>
145    * {@linkplain #expect Expectations} are supported via Hamcrest matchers.
146    * <p>
147    * To obtain all output to {@code System.out} and {@code System.err}, simply invoke {@link #toString()}.
148    *
149    * @author Phillip Webb
150    * @author Andy Wilkinson
151    * @author Sam Brannen
152    */
153   static class OutputCapture {
154 
155     private final List<Matcher<? super String>> matchers = new ArrayList<>();
156 
157     private CaptureOutputStream captureOut;
158 
159     private CaptureOutputStream captureErr;
160 
161     private ByteArrayOutputStream copy;
162 
163     void captureOutput() {
164       this.copy = new ByteArrayOutputStream();
165       this.captureOut = new CaptureOutputStream(System.out, this.copy);
166       this.captureErr = new CaptureOutputStream(System.err, this.copy);
167       System.setOut(new PrintStream(this.captureOut));
168       System.setErr(new PrintStream(this.captureErr));
169     }
170 
171     void releaseOutput() {
172       System.setOut(this.captureOut.getOriginal());
173       System.setErr(this.captureErr.getOriginal());
174       this.copy = null;
175     }
176 
177     private void flush() {
178       try {
179         this.captureOut.flush();
180         this.captureErr.flush();
181       } catch (IOException ex) {
182         // ignore
183       }
184     }
185 
186     /**
187      * Verify that the captured output is matched by the supplied {@code matcher}.
188      * <p>
189      * Verification is performed after the test method has executed.
190      *
191      * @param matcher
192      *          the matcher
193      */
194     public void expect(Matcher<? super String> matcher) {
195       this.matchers.add(matcher);
196     }
197 
198     /**
199      * Return all captured output to {@code System.out} and {@code System.err} as a single string.
200      */
201     @Override
202     public String toString() {
203       flush();
204       return this.copy.toString();
205     }
206 
207     void reset() {
208       this.matchers.clear();
209       this.copy.reset();
210     }
211 
212     private static class CaptureOutputStream extends OutputStream {
213 
214       private final PrintStream original;
215 
216       private final OutputStream copy;
217 
218       CaptureOutputStream(PrintStream original, OutputStream copy) {
219         this.original = original;
220         this.copy = copy;
221       }
222 
223       PrintStream getOriginal() {
224         return this.original;
225       }
226 
227       @Override
228       public void write(int b) throws IOException {
229         this.copy.write(b);
230         this.original.write(b);
231         this.original.flush();
232       }
233 
234       @Override
235       public void write(byte[] b) throws IOException {
236         write(b, 0, b.length);
237       }
238 
239       @Override
240       public void write(byte[] b, int off, int len) throws IOException {
241         this.copy.write(b, off, len);
242         this.original.write(b, off, len);
243       }
244 
245       @Override
246       public void flush() throws IOException {
247         this.copy.flush();
248         this.original.flush();
249       }
250 
251     }
252 
253   }
